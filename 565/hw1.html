2.1.1 Random Sample from Bivariate Normal Distribution¶

$$Let \ X \sim N_2 \biggl(\mu = \left[ {\begin{array}{c} 0\\ 0\\ \end{array} } \right],\Sigma = \left[ {\begin{array}{cc} 9 & \sigma_{12} \\ \sigma_{21} & 4 \\ \end{array} } \right] \biggl) $$

We are going to select three random samples(say S1, S2, S3) size n from X with correlation 0, 0.5 and 9 and discuss their shapes, how it affected by correlation coefficient.

We know;

$$ \sigma_{12} = \sigma_{21} = \rho_{12}\sqrt{\sigma_{22}\sigma_{11}} $$

for S1; $$ \sigma_{12} = \sigma_{21} = 0 \times \sqrt{9 \times 4 } = 0$$

for S2; $$ \sigma_{12} = \sigma_{21} = 0.5 \times \sqrt{9 \times 4 } = 3$$

for S3; $$ \sigma_{12} = \sigma_{21} = 0.9 \times \sqrt{9 \times 4 } = 5.4$$

sample_BVN = function(n, M, V, seed = 108){
  
  # n = (scaler int) sample size
  # M = (size 2 vector) mean vector
  # V = (size 2 vector) variance vector
  
  function(correlation){
      
    # correlation  = (scaler float) desired correlation coefficient
  
    set.seed(seed)
    
    M = matrix(M, 2, 1)
    S = matrix(c(V[1], correlation*sqrt(V[1]*V[2]), correlation*sqrt(V[1]*V[2]), V[2]), 2, 2)
    
    X = rmvnorm(n, M, S)
    z = apply(X, 1, function(x) dmvnorm(x, M, S))
    
    return(cbind(X,z))
  }
}

M = c(0, 0)
V = c(9, 4)
n = 1000

#Using closure feature of R, we create sample_BVNX function with predefined n, M and V. It only needs correlation input:
sample_BVNX = sample_BVN(n, M, V)

contour_histogram = function(data){
  
  # data = (n x 3 matrix) sampled bivariate normal data and their density value
  
  x1 = data[,1]
  x2 = data[,2]
  s <- subplot(
    plot_ly(x = x1, type = "histogram"),
    plotly_empty(),
    plot_ly(x = x1, y = x2, type = "histogram2dcontour"),
    plot_ly(y = x2, type = "histogram"),
    nrows = 2, heights = c(0.2, 0.8), widths = c(0.8, 0.2), margin = 0,
    shareX = TRUE, shareY = TRUE, titleX = FALSE, titleY = FALSE
  )
  layout(s, showlegend = FALSE)
}

plot_mvn = function(M, S, intervalx1, intervalx2){

  # M = (size 2 vector) means of marginals
  # S = (2 x 2 matrix) parameter of first random variable
  # intervalx1 = (size 2 vector) desired x1 interval for plotting
  # intervalx2 = (size 2 vector) desired x2 interval for plotting
  
  x1 = seq(intervalx1[1], intervalx1[2], length = 100)
  x2 = seq(intervalx2[1], intervalx2[2], length = 100)
  Mu = mesh(x1, x2)
  
  x = Mu$x
  y = Mu$y
  
  z = matrix(0, 100, 100)
  for(i in 1:100){
    for(j in 1:100){
      z[i, j] = dmvnorm(x = c(x[i, j], y[i, j]), M, S) 
    }
  }
  
  df = list(x1 = x1, x2 = x2, z=z)
  
  plot_ly(x = df$x1, y = df$x2, z = df$z) %>% add_surface()
}

dBVP = function(x, theta1, theta2, theta0){
  
  # x = (size 2 vector) x1 and x2 values
  # theta1 = (scaler) parameter of first random variable
  # theta2 = (scaler) parameter of second random variable
  # theta0 = (scaler) parameter of covariance of random variable
  
  u = x[1]
  v = x[2]
  
  a = exp(-(theta1+theta2+theta0))
  b = ((theta1^u)*(theta2^v))/(factorial(u)*factorial(v))

  c = 0
  i_from = 0
  i_to = min(u,v)
  for(i in i_from:i_to){
    c1 = dim(combn(u,i))[2]*dim(combn(v,i))[2]
    c2 = factorial(i)*((theta0/(theta1*theta2))^i)
    c = c + c1*c2
  }
  
  indicator = (theta1 > theta0)*(theta2 > theta0)*(theta0 > 0)
  
  return(a*b*c*indicator)
  
}

BVP_data = function(theta1, theta2, intervalx1, intervalx2){
    
  # theta1 = (scaler) parameter of first random variable
  # theta2 = (scaler) parameter of second random variable
  # intervalx1 = (size n integer vector) desired x1 interval for plotting
  # intervalx2 = (size n integer vector) desired x2 interval for plotting
  
  function(correlation){
      
  # correlation  = (scaler float) desired correlation coefficient
    
    theta0 = correlation*sqrt(theta1*theta2)
    
    bvpdat = expand.grid(intervalx1, intervalx2)
    bvpdat = data.frame(bvpdat)
    colnames(bvpdat) = c("x1", "x2")
    bvpdat$density = apply(bvpdat, 1, function(x) dBVP(x, theta1, theta2, theta0))
    
    return(bvpdat)
  }
}

We got similar results with continous cases(It is quite similar to contour). The only difference is, when we increase covariance parameter, mean of both marginals also increased. This is the characteristic of Poisson distribution. In Normal Distribution, mean and variances are independent. They have their own, independent parameters. However, it dependes same parameters in Poissson. Spesifically, in bivariate poisson marginal distributions are as follows;

$$ X_1 \sim Poisson(\theta_1+\theta_0) \\ X_2 \sim Poisson(\theta_2+\theta_0) $$

As a result, not only the shape of distribution changes, location also shifts.
