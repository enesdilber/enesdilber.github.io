rBVN = function(M, V, cor = 0, seed = 108){
    
    #M = (vector of size 2) Mean of BVN
    #V = (vector of size 2) Variance of BVN
    #cor = (scaler between 0 and 1) Correlation coefficient of X1 and X2
    #seed = (scaler) random seed
    
    M = matrix(M, 2, 1)
    s11 = V[1]
    s22 = V[2]
    s12 = cor*sqrt(s11*s22)
    S = matrix(c(s11, s12, s12, s22), 2, 2)
    
    function(n){
        
        #n = (scaler integer) sample size
        
        set.seed(seed)
        
        return(rmvnorm(n = n, mean = M, sigma = S))
    }
}

3 Likelihood Function¶

We know if $ X_i \sim iid \ N_2(\mu, \Sigma)$ then;

$$ L(X; \mu, \Sigma) = \prod{f(x_i; \mu, \Sigma) } $$

To find MLE of $\mu$ and $\Sigma$ we need to find;

$$ \widehat{\mu} = argmax \ L(X; \mu, \Sigma) \ w.r.t. \ \mu \\$$ $$ \widehat{\Sigma} = argmax \ L(X; \mu, \Sigma) \ w.r.t. \ \Sigma $$

We are going to directly implement Log-Leklihood as a function:

Lik_BVN = function(X){
  
  # X = (matrix of size n x 2) sampled data
  n = dim(X)[1]
  x_bar = colMeans(X)
  
  function(Theta){
    
    
    M = matrix(Theta[1:2], 2, 1)
    S = matrix(Theta[c(3, 5, 5, 4)], 2, 2)
    Sinv = solve(S)
    
    temp = matrix(0, 2, 2)
    
    for(i in 1:n){
      temp = temp + (X[i, ] - x_bar) %*% t(X[i, ] - x_bar)
    }
    
    A = -n*log(2*pi) - (n/2)*log(det(S))
    
    B = Sinv%*%temp
    
    C = n*(t(x_bar - M) %*% Sinv %*% (x_bar - M))
    
    
    result = A-0.5*(sum(diag(B)) + C)
    
    return(result)
    
  }
}

3 Calculating Numerical Gradient¶

We are going differentiate and find the Gradient by implementing the limit definition of derivative.

$$ f'(x) = \lim_{h \to 0} \frac{f(x+h) - f(x-h)}{2h}$$

Here is the numerical gradient function:

In [5]:
NG = function(f, x, epsilon = 1e-4){
    
    # f (function) function of interest
    # x (vector or scaler) f'(x) point of gradient
    # epsilon (scaler) small constant    
    
    n = length(x)
    
    GradX = numeric()
    
    for(i in 1:n){

        h = rep(0, n)
        h[i] = epsilon

        x1 = x + h
        x0 = x - h

        GradX[i] =  (f(x1) - f(x0))/(2*epsilon)
    }
    return(GradX)
}

4 Optimization with Numerical Gradient Decsent¶

NGD = function(f, Theta, lr = 10e-5, iter = 100, dec = 3){
    
    # f = function to be maximized
    # initial = (vector) initial points for the function
    # lr = (float) learning rate
    # iter = (integer) iteration number
    # dec = (integer) number of significant decimals for zero value of gradient
    
    LIK = numeric(0)
    Theta_M = Theta 
    
    ptm <- proc.time()

    for(i in 1:iter){
        
        GradTheta = NG(f, Theta)        
        Theta = Theta + lr*GradTheta
        Theta_M = rbind(Theta_M, Theta)
        LIK[i] = f(Theta)
        
        if(round(max(abs(GradTheta)), dec) == 0){
            
            
            print(paste("iteration has ended at step", i))  
            print(paste("Elapsed Time is", (proc.time() - ptm)[3], "seconds"))
            
            return(list(Likelihood = LIK, Theta = Theta_M))
            
        } 
    }
    
    print(paste("Elapsed Time is", (proc.time() - ptm)[3], "seconds"))
        
    return(list(Likelihood = LIK, Theta = Theta_M))
}

5 Results¶

Here is the convergence behavior of Log-Likelihood function with respect to sample size. One can see that, at same starting point, number of iterations x step size(learning rate) decrease by sample size. This is expected, because in our algorithm, for initial point we select the population $\theta$ for all samples. And we know when sample size increases, behavior of sample converges to population. Clearly, we could conclude that if we select a sample size n, where n is a very large number, iterations would end up at first steps. However, one could mind the time complexity.

calc_last = function(x){
  
  n = dim(x)[1]
  max1 = which(round(x[,1], 5) == round(x[n,1], 5))[1]
  max2 = which(round(x[,2], 5) == round(x[n,2], 5))[1]
  max(max1, max2)
}

calc_mus = function(mus, name){

  n = calc_last(mus)
  mus = mus[1:n,]
  rownames(mus) = NULL
  name = rep(name, n)
  mus = data.frame(name, mus)
  colnames(mus) = c("sample_size", "mu1", "mu2")
  return(mus)
}
